## Given a string .Check if the string is palindrome or not?
bool isPlaindrome(string str) {
	int n = str.length();
	for(int i = 0, j = n - 1; i < j; ++i, --j) {
		if(str[i] != str[j]) return false;
	}
        return true;
}
